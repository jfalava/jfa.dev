import { isLocaleInList, supportedLanguages } from 'src/components/get-locale.ts';
export default function HeaderTitle() {
  const [translatedTitle, setTranslatedTitle] = createSignal("mdx");

  if (isLocaleInList(supportedLanguages)) {
    const userLocale = supportedLanguages[0].toLowerCase();
    switch (userLocale) {
      case "es-es":
      case "es-419":
      case "es":
        setTranslatedTitle("spanish text");
        break;
      case "en-us":
      case "en-gb":
      case "en":
        setTranslatedTitle("english text");
        break;
      default:
        setTranslatedTitle("default text");
    }
  }
}
